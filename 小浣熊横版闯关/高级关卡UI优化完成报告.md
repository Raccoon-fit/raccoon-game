### JavaScript控制增强
```javascript
// 虚拟控制支持
setupVirtualControls() {
    // 触摸事件处理
    btn.addEventListener('touchstart', (e) => {
        e.preventDefault();
        this.keys[key] = true;
    });
    
    // 触摸事件结束
    btn.addEventListener('touchend', (e) => {
        e.preventDefault();
        this.keys[key] = false;
    });
}
```
