## 💻 技术实现

### 简单重置函数
```javascript
function simpleReset() {
    console.log('🔄 执行简单重置...');
    
    // 清除高级关卡数据
    localStorage.removeItem('raccoon_adventure_advanced_progress');
    
    alert('重置完成！页面将重新加载以应用更改。');
    setTimeout(() => {
        window.location.reload();
    }, 500);
}
```

### 清除所有数据函数
```javascript
function clearAllData() {
    if (confirm('确定要清除所有游戏数据吗？这将删除所有进度和设置！')) {
        // 清除所有可能的数据
        localStorage.removeItem('raccoon_adventure_advanced_progress');
        localStorage.removeItem('raccoon_adventure_progress');
        localStorage.removeItem('raccoon_adventure_advanced_best_time');
        
        alert('所有数据已清除！页面将重新加载。');
        setTimeout(() => {
            window.location.reload();
        }, 500);
    }
}
```

### 状态显示函数
```javascript
function showStatus() {
    const data = localStorage.getItem('raccoon_adventure_advanced_progress');
    if (data) {
        const progress = JSON.parse(data);
        // 显示详细状态信息
        alert(`当前进度：
- 最大解锁关卡: ${progress.maxUnlockedLevel || 6}
- 关卡星星数据...`);
    } else {
        alert('没有找到保存的进度数据。');
    }
}
```
